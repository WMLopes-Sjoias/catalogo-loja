Gostaria de encomendar estes itens:\n"; mensagem += "──────────────────\n\n"; this.itens.forEach(item => { mensagem += `🔸 *${item.nome}*\n`; mensagem += ` Qtd: ${item.quantidade} x R$ ${item.preco}\n`; mensagem += ` Cód: ${item.codigo}\n\n`; }); mensagem += "──────────────────\n"; mensagem += `💰 *Total: R$ ${this.total.toFixed(2).replace('.', ',')}*`; return mensagem; } }; // Inicializa o total ao carregar window.carrinho.calcularTotal();
