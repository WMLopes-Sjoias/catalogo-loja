function doGet(e) { // Verifica se recebeu os parâmetros if (!e || !e.parameter) { return ContentService.createTextOutput("❌ Parâmetros não informados"); } var total = e.parameter.total; var descricao = e.parameter.descricao || "Compra no site WMLopes"; if (!total) { return ContentService.createTextOutput("❌ Total não informado"); } // 🔥 SEU TOKEN DE PRODUÇÃO var token = "02540c13-e53a-4c0e-a1d4-d5da4e609548199a04284b1f977a64359faa71af8d1e64e9-3d7e-4ae5-806d-f74c8603f106"; var email = "wilsonlopesjr@hotmail.com"; // Formata o total (usa ponto, não vírgula) var totalFormatado = parseFloat(total).toFixed(2); // Dados do pagamento var dados = { email: email, token: token, currency: "BRL", itemId1: "1", itemDescription1: descricao.substring(0, 100), itemAmount1: totalFormatado, itemQuantity1: "1", itemWeight1: "0", reference: "pedido_" + new Date().getTime(), redirectURL: "https://WMLopes-Sjoias.github.io/catalogo-loja/obrigado.html", notificationURL: "" }; // Converte para formato URL-encoded var payload = Object.keys(dados).map(key => key + "=" + encodeURIComponent(dados[key])).join("&"); var options = { method: "post", payload: payload, muteHttpExceptions: true }; // 🔥 URL DE PRODUÇÃO (PagSeguro real) var url = "https://ws.pagseguro.uol.com.br/v2/checkout"; try { var response = UrlFetchApp.fetch(url, options); var xml = response.getContentText(); // Extrai o código do checkout var codeMatch = xml.match(/(.*?)<\/code>/); if (codeMatch && codeMatch[1]) { var checkoutCode = codeMatch[1]; var linkPagamento = "https://pagseguro.uol.com.br/v2/checkout/payment.html?code=" + checkoutCode; return ContentService.createTextOutput(linkPagamento); } else { // Se não encontrou código, retorna o XML completo para debug return ContentService.createTextOutput("❌ Erro: código não encontrado. Resposta: " + xml); } } catch (err) { return ContentService.createTextOutput("❌ Erro na integração: " + err.toString()); } }
